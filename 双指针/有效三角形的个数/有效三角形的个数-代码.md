class Solution {
public:
    int triangleNumber(vector<int>& nums) {
        //排序
        sort(nums.begin(), nums.end());
        //计算三元组个数
        int count = 0;
        for (int end = nums.size() - 1; end > 1; --end)
        {
            int left = 0; 
            int right = end - 1;
            while (left < right)
            {
                if (nums[left] + nums[right] > nums[end])
                {
                    count += right - left;
                    --right;
                }
                else
                {
                    ++left;
                }
            }
        }
        return count; 
    }
};
